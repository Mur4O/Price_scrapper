<div class="desc p">The GeForce RTX 5090 is an enthusiast-class graphics card by NVIDIA, launched on January 30th, 2025. Built on the 5 nm process, and based on the GB202 graphics processor, in its GB202-300-A1 variant, the card supports DirectX 12 Ultimate. This ensures that all modern games will run on GeForce RTX 5090. Additionally, the DirectX 12 Ultimate capability guarantees support for hardware-raytracing, variable-rate shading and more, in upcoming video games. The GB202 graphics processor is a large chip with a die area of 750 mm² and 92,200 million transistors. It features 21760 shading units, 680 texture mapping units,  and 176 ROPs. Also included are 680 tensor cores which help improve the speed of machine learning applications. The card also has 170 raytracing acceleration cores. NVIDIA has paired 32 GB GDDR7 memory with the GeForce RTX 5090, which are connected using a 512-bit memory interface. The GPU is operating at a frequency of 2017 MHz, which can be boosted up to 2407 MHz, memory is running at 1750 MHz (28 Gbps effective). <br /> Being a dual-slot card, the NVIDIA GeForce RTX 5090 draws power from 1x 16-pin power connector, with power draw rated at 575 W maximum. Display outputs include: 1x HDMI 2.1b, 3x DisplayPort 2.1b. GeForce RTX 5090 is connected to the rest of the system using a PCI-Express 5.0 x16 interface. The card's dimensions are 304 mm x 137 mm x 40 mm, and it features a dual-slot cooling solution.  Its price at launch was 1999 US Dollars.</div>

				$('.customboards').find('td').filter(function() {
					return $(this).index() == idx;

					return $(a).data('sort') > $(b).data('sort') ? inverse ? -1 : 1 : inverse ? 1 : -1;
				}, function() {
					return this.parentNode;
				});
			}

    	<script>
	(function() {
		// State: 1=loaded, 2=viewed, 3=engaged
		const data = { e: {} }
		const sentState = {}
		const durations = {}
		let observer
		let flushCount = 0
		let bfcacheRestore = false
		const loadTime = Date.now()
		const lsKey = '_adst'
		let hasLS = false

		try { localStorage.setItem('_t', '1'); localStorage.removeItem('_t'); hasLS = true } catch(e) {}

		function lsRead() {
			if (!hasLS) return {}
			try {
				var raw = localStorage.getItem(lsKey)
				if (!raw) return {}
				var parsed = JSON.parse(raw)
				var now = Date.now()
				var result = {}
				for (var id in parsed) {
					if (now - parsed[id].t < 1800000) result[id] = parsed[id].s
				}
				return result
			} catch(e) { return {} }
		}

		function lsWrite() {
			if (!hasLS) return
			try {
				var existing = {}
				try { var raw = localStorage.getItem(lsKey); if (raw) existing = JSON.parse(raw) } catch(e) {}
				var now = Date.now()
				for (var id in sentState) existing[id] = { s: sentState[id], t: now }
				// cleanup old entries
				for (var id in existing) {
					if (now - existing[id].t >= 1800000) delete existing[id]
				}
				localStorage.setItem(lsKey, JSON.stringify(existing))
			} catch(e) {}
		}

		window.addEventListener('pageshow', function(e) {
			if (e.persisted) bfcacheRestore = true
		})

		function flush(reason) {
			var toSend = {}
			for (var id in data.e) {
				if (!sentState[id] || data.e[id] > sentState[id]) {
					toSend[id] = data.e[id]
				}
			}

			if (!Object.keys(toSend).length) return

			for (var id in toSend) sentState[id] = toSend[id]
			lsWrite()

			flushCount++
			var tokens = Object.keys(toSend).join(',')
			var age = Math.round((Date.now() - loadTime) / 1000)
			var qs = '?t=' + tokens + '&r=' + reason + '&n=' + flushCount + '&bf=' + (bfcacheRestore ? 1 : 0) + '&age=' + age + '&vis=' + document.visibilityState
			navigator.sendBeacon('/_confirm-loaded' + qs, JSON.stringify({ e: toSend }))
		}

		function onPageHide() { flush('pagehide') }

		function onVisibilityChange() {
			if (document.visibilityState === 'hidden' && Object.values(data.e).every(v => v === 3)) flush('vischange')
		}

		function init() {
			const els = document.querySelectorAll('[data-req-id]')
			if (!els.length) return

			var stored = lsRead()
			els.forEach(el => {
				var id = el.dataset.reqId
				data.e[id] = 1
				if (stored[id]) sentState[id] = stored[id]
			})

			if (!('IntersectionObserver' in window)) {
				data.io = false
				flush('no-io')
				return
			}

			observer = new IntersectionObserver(entries => {
				entries.forEach(({ target, isIntersecting, intersectionRatio }) => {
					const id = target.dataset.reqId

					if (isIntersecting && data.e[id] < 2) data.e[id] = 2

					if (intersectionRatio >= 0.5) {
						if (!durations[id]) {
							durations[id] = setTimeout(() => {
								data.e[id] = 3
								if (Object.values(data.e).every(v => v === 3)) flush('engaged')
							}, 1000)
						}
					} else {
						clearTimeout(durations[id])
						durations[id] = null
					}
				})
			}, { threshold: [0, 0.5] })

			els.forEach(el => observer.observe(el))

			document.addEventListener('visibilitychange', onVisibilityChange)
			window.addEventListener('pagehide', onPageHide)

			setInterval(function() { flush('periodic') }, 600000)
		}

		if (document.readyState === 'loading') {
			document.addEventListener('DOMContentLoaded', init)
		} else {
			init()
		}
	})()
	</script>
